CREATE TABLE IF NOT EXISTS SINCEID(id serial PRIMARY KEY, name varchar, value bigint DEFAULT 0);
CREATE TABLE IF NOT EXISTS TASKS(id serial PRIMARY KEY, name varchar, data varchar,  completed boolean, tstamp TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(), CHECK(EXTRACT(TIMEZONE FROM tstamp) = '0'));
CREATE TABLE IF NOT EXISTS FOLLOWERS(id serial PRIMARY KEY, screen_name varchar);
CREATE TABLE IF NOT EXISTS FOLLOWS(id serial PRIMARY KEY, screen_name varchar);
